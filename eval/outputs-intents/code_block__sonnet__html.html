import time
import random
import urllib.request
import urllib.error
from typing import Optional

# Errors worth retrying on
RETRYABLE_CODES = {429, 500, 502, 503, 504}


def fetch_with_backoff(
    url: str,
    *,
    max_retries: int = 5,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    jitter: bool = True,
    timeout: Optional[float] = 10.0,
) -> bytes:
    """
    Fetch *url* with exponential-backoff retries.

    Args:
        url         – Target URL.
        max_retries – Maximum number of retry attempts.
        base_delay  – Initial wait in seconds (doubles each round).
        max_delay   – Upper ceiling on wait time.
        jitter      – Add randomness to avoid thundering herd.
        timeout     – Per-request socket timeout in seconds.

    Returns:
        Raw response body as bytes.

    Raises:
        urllib.error.URLError on permanent failure or exhausted retries.
    """
    attempt = 0

    while True:
        try:
            with urllib.request.urlopen(url, timeout=timeout) as resp:
                return resp.read()

        except urllib.error.HTTPError as exc:
            # Don't retry client errors (4xx) except 429
            if exc.code not in RETRYABLE_CODES or attempt >= max_retries:
                raise
            last_exc = exc

        except urllib.error.URLError as exc:
            # Network-level failure — always retry
            if attempt >= max_retries:
                raise
            last_exc = exc

        # --- Backoff math ---
        delay = min(base_delay * (2 ** attempt), max_delay)
        if jitter:
            delay *= random.uniform(0.5, 1.5)

        print(
            f"Attempt {attempt + 1} failed "
            f"({last_exc}). "
            f"Retrying in {delay:.1f}s…"
        )
        time.sleep(delay)
        attempt += 1


# ── Quick demo ────────────────────────────────────────────────────────────
if __name__ == "__main__":
    body = fetch_with_backoff(
        "https://httpbin.org/get",
        max_retries=4,
        base_delay=0.5,
    )
    print(f"Got {len(body)} bytes")
