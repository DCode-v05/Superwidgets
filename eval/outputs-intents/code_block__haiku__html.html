import requests
import time
from typing import Optional

def fetch_with_backoff(
    url: str,
    max_retries: int = 4,
    base_delay: float = 1.0,
    max_delay: float = 60.0,
    timeout: int = 10,
) -> Optional[requests.Response]:
    """
    Fetch URL with exponential backoff retry strategy.
    
    Args:
        url: Target URL
        max_retries: Number of retry attempts (4 = 4 failures before giving up)
        base_delay: Initial delay in seconds (1.0)
        max_delay: Cap on any single delay (60.0)
        timeout: Request timeout in seconds
    
    Returns:
        Response object on success, None on final failure
    """
    
    for attempt in range(max_retries + 1):
        try:
            response = requests.get(url, timeout=timeout)
            response.raise_for_status()
            return response
        
        except (requests.RequestException, requests.Timeout) as e:
            if attempt == max_retries:
                # Final attempt failed
                print(f"Failed after {max_retries + 1} attempts: {e}")
                return None
            
            # Exponential backoff with jitter
            delay = min(
                base_delay * (2 ** attempt),
                max_delay
            )
            # Add random jitter (±20%)
            jitter = delay * 0.2 * (2 * __import__('random').random() - 1)
            actual_delay = delay + jitter
            
            print(f"Attempt {attempt + 1} failed. Retrying in {actual_delay:.2f}s...")
            time.sleep(actual_delay)

# Usage
response = fetch_with_backoff("https://api.example.com/data")
if response:
    data = response.json()
Key features:
• Exponential 2ⁿ backoff (1s → 2s → 4s → 8s → 16s)
• Jitter (±20%) to avoid thundering herd
• Configurable max delay cap
• Catches timeouts + connection errors
• Returns None on exhaustion
