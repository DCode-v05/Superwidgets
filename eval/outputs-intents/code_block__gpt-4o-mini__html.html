Exponential Backoff URL Fetcher

import time
import requests

def fetch_with_retries(url, max_retries=5):
    retries = 0
    backoff_time = 1  # Start with a 1 second backoff
    
    while retries < max_retries:
        try:
            response = requests.get(url)
            response.raise_for_status()  # Raise an error for bad responses
            return response.text
        except requests.RequestException as e:
            print(f"Error occurred: {e}. Retrying in {backoff_time}s...")
            time.sleep(backoff_time)
            backoff_time *= 2  # Double the backoff time
            retries += 1
            
    raise Exception("Maximum retries exceeded")
